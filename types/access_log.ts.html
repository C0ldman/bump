export function mapAccessLog(log: NetAccessLog): AccessLog {
  return {
    uuid: log.uuid,
    accessDescription: mapAccessDescription(log.access_description),
    time: new Date(log.time),
    doorUuid: log.door_uuid,
    sectionUuid: log.section_uuid,
    siteUuid: log.site_uuid,
    userUuid: log.user_uuid,
    pincode: log.pincode,
    direction: mapDirection(log.direction),
    authType: mapAuthTypeToEnum(log.auth_type),
    reason: log.reason || '',
    userEmail: log.user_email,
    userName: log.user_name
  };
}
 
export function mapPbAccessLog(log: PbAccessLog): AccessLog {
  const time = log.getTime();
  const timeVal = time ? time.toDate() : null;
  return {
    uuid: log.getUuid(),
    accessDescription: log.getAccessDescription(),
    time: timeVal,
    doorUuid: log.getDoorUuid(),
    sectionUuid: log.getSectionUuid(),
    siteUuid: log.getSiteUuid(),
    userUuid: log.getUserUuid(),
    pincode: log.getPincodeValue(),
    direction: log.getDirection(),
    authType: log.getAuthType(),
    reason: log.getReason(),
    userEmail: log.getUserEmail(),
    userName: log.getUserName()
  };
}

export function mapMonthDayStat(stat: MonthlyAccessCountResponse): AccessCount {

  };
}
 
function mapAccessDescription(ad: string): AccessDescription {
  switch (ad) {
    case 'access_granted': return AccessDescription.ACCESS_DESCRIPTION_ACCESS_GRANTED;
    case 'access_denied': return AccessDescription.ACCESS_DESCRIPTION_ACCESS_DENIED;
    default: return AccessDescription.ACCESS_DESCRIPTION_UNSPECIFIED;
  }
}
 
function mapDirection(direction: string): Direction {
  switch (direction) {
    case 'in': return Direction.DIRECTION_IN;
    case 'out': return Direction.DIRECTION_OUT;
    default: return Direction.DIRECTION_UNDEFINED;
  }
}

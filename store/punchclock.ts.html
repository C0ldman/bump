import { createErrorList } from '@/utils/helper-functions';
import { MutationTree, ActionTree, GetterTree } from "vuex";
import {RootState, mapPbIndoorSession} from '@/types/indoor_session';
import {GetIndoorSessionsRequest, GetIndoorSessionsResponse} from "@/entryfy/org/api/v2/indoor_sessions_service_pb";
import {IndoorSessionsServiceClient} from "@/entryfy/org/api/v2/Indoor_sessions_serviceServiceClientPb";
 
const GET_INDOOR_SESSIONS_PENDING = 'GET_INDOOR_SESSIONS_PENDING';
const GET_INDOOR_SESSIONS_SUCCESS = 'GET_INDOOR_SESSIONS_SUCCESS';
const GET_INDOOR_SESSIONS_FAILURE = 'GET_INDOOR_SESSIONS_FAILURE';
 
const CLEAR_INDOOR_SESSIONS = 'CLEAR_INDOOR_SESSIONS';
 
export const state = (): RootState => ({
  pending: false,
  indoorSessions: []
});
 
export const getters: GetterTree<RootState, RootState> = {
  indoorSessions: (state) => {
    return state.indoorSessions;
  },
  isPending: (state) => {
    return state.pending;
  }
};
 
export const mutations: MutationTree<RootState> = {
  [GET_INDOOR_SESSIONS_PENDING](state) {
    state.pending = true;
  },
  [GET_INDOOR_SESSIONS_SUCCESS](state, payload: GetIndoorSessionsResponse) {
    state.indoorSessions = payload.getIndoorSessionsList().map((session) => mapPbIndoorSession(session));
    state.pending = false;
  },
  [GET_INDOOR_SESSIONS_FAILURE](state, _payload) {
    state.pending = false;
  },
  [CLEAR_INDOOR_SESSIONS](state) {
    state.indoorSessions = [];
  }
};
 
 
export const actions: ActionTree<RootState, RootState> = {
  async getIndoorSessions({ commit }, body: GetIndoorSessionsRequest) {
    commit(GET_INDOOR_SESSIONS_PENDING);
    try {
      const indoorSessionsServiceClient =
        new IndoorSessionsServiceClient(process.env.accountGRPCUrl || 'localhost:8080');
      const token = localStorage.getItem('token');
 
      const result = await indoorSessionsServiceClient.getIndoorSessions(body, { Authorization: token || '' });
 
      commit(GET_INDOOR_SESSIONS_SUCCESS, result);
      return result;
    } catch (e) {
      commit(GET_INDOOR_SESSIONS_FAILURE, 'fetching indoor sessions data failed');
      throw createErrorList(e);
    }
  },
 
  clear({ commit }) {
    commit(CLEAR_INDOOR_SESSIONS);
  }
};
